  <script>
    function toggleForm(formId) {
      document.getElementById('loginForm').style.display = 'none';
      document.getElementById('signupForm').style.display = 'none';
      document.getElementById('portal').style.display = 'none';
      document.getElementById(formId).style.display = 'block';
    }

    function signup() {
      let name = document.getElementById('signupName').value;
      let email = document.getElementById('signupEmail').value;
      let password = document.getElementById('signupPassword').value;

      if(name && email && password) {
        localStorage.setItem('studentName', name);
        localStorage.setItem('studentEmail', email);
        localStorage.setItem('studentPassword', password);
        alert("Signup successful! Please login.");
        toggleForm('loginForm');
      } else {
        alert("Please fill all fields.");
      }
    }

    function login() {
      let email = document.getElementById('loginEmail').value;
      let password = document.getElementById('loginPassword').value;

      let storedEmail = localStorage.getItem('studentEmail');
      let storedPassword = localStorage.getItem('studentPassword');
      let storedName = localStorage.getItem('studentName');

      if(email === storedEmail && password === storedPassword) {
        document.getElementById('studentName').innerText = storedName;
        document.getElementById('studentEmail').innerText = storedEmail;
        toggleForm('portal');
      } else {
        alert("Invalid email or password.");
      }
    }

    function logout() {
      toggleForm('loginForm');
    }
  </script>
